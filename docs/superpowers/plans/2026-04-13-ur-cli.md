# ur-cli Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build `ur-cli`, a ratatui terminal frontend for the Royal Game of Ur, as a peer crate to `ur-core`.

**Architecture:** `App` owns all UI state and a `Screen` enum that drives rendering and input routing. All game logic is delegated to `ur-core` — the CLI never computes moves or win conditions itself. Animations are frame-based and non-blocking, advancing on each tick of the event loop.

**Tech Stack:** Rust, ratatui 0.26, crossterm 0.27, rand 0.8, ur-core (workspace path dep)

/// Which screen is currently active.
#[derive(Debug)]
pub enum Screen {
    Title,
    DifficultySelect { selected: usize },
    DiceOff { state: DiceOffState },
    Game,
    GameOver,

/// State for the first-player dice-off animation.
#[derive(Debug)]
pub struct DiceOffState {
    /// Frames remaining in the animation for each player's dice.
    pub p1_frames: u32,
    pub p2_frames: u32,
    /// Final rolled values (set at animation start).
    pub p1_final: Dice,
    pub p2_final: Dice,
    /// Currently displayed values (cycle during animation).
    pub p1_display: Dice,
    pub p2_display: Dice,
    /// Set once animation finishes and winner is determined.
    pub winner: Option<ur_core::player::Player>,
    /// Whether the user has acknowledged the result.
    pub acknowledged: bool,
}

pub const DIFFICULTIES: [(&str, u32); 3] = [
    ("Easy", 2),
    ("Medium", 4),
    ("Hard", 6),
];

    pub moves: u32,
    pub start_time: Option<Instant>,

    pub log_visible: bool,

    pub rng: SmallRng,
}

impl App {
    /// Creates a new `App` starting on the title screen.
    pub fn new() -> Self {
        Self {
            screen: Screen::Title,
            should_quit: false,
            difficulty: 4, // Medium default
            game_state: None,
            dice_roll: None,
            legal_moves: Vec::new(),
            selected_move_idx: 0,
            log: Vec::new(),
            log_visible: false,
            stats: GameStats::default(),
            rng: SmallRng::from_entropy(),
        }
    }

    /// Transitions to difficulty selection.
    pub fn start_new_game(&mut self) {
        self.screen = Screen::DifficultySelect { selected: 1 }; // default Medium
    }

    /// Reads selected difficulty, transitions to dice-off.
    pub fn confirm_difficulty(&mut self) {
        let selected = match self.screen {
            Screen::DifficultySelect { selected } => selected,
            _ => 1,
        };
        self.difficulty = DIFFICULTIES[selected].1;

        let p1_final = Dice::roll(&mut self.rng);
        let p2_final = Dice::roll(&mut self.rng);
        self.screen = Screen::DiceOff {
            state: DiceOffState {
                p1_frames: 18, // ~600ms at 30fps
                p2_frames: 18,
                p1_final,
                p2_final,
                p1_display: Dice(0),
                p2_display: Dice(0),
                winner: None,
                acknowledged: false,
            },
        };
    }

    /// Sets `should_quit = true`.
    pub fn quit(&mut self) {
        self.should_quit = true;
    }
}

```rust
use crossterm::event::{KeyCode, KeyEvent};
use crate::app::Screen;

/// High-level actions derived from raw key events.
#[derive(Debug, PartialEq, Eq)]
pub enum Action {
    // Navigation (menus and title)
    MenuUp,
    MenuDown,
    Confirm,
    Back,
    // Gameplay
    RollDice,
    SelectPrev,
    SelectNext,
    ConfirmMove,
    ToggleLog,
    QuitPrompt,
    // Game over
    NewGame,
    Quit,
}

/// Maps a raw key event to an `Action` given the current screen.
/// Returns `None` for keys that have no binding in this context.
pub fn map_key(key: KeyEvent, screen: &Screen) -> Option<Action> {
    match screen {
        Screen::Title => match key.code {
            KeyCode::Up | KeyCode::Char('k') => Some(Action::MenuUp),
            KeyCode::Down | KeyCode::Char('j') => Some(Action::MenuDown),
            KeyCode::Enter | KeyCode::Char(' ') => Some(Action::Confirm),
            KeyCode::Char('q') | KeyCode::Esc => Some(Action::Quit),
            _ => None,
        },
        Screen::DifficultySelect { .. } => match key.code {
            KeyCode::Up | KeyCode::Char('k') => Some(Action::MenuUp),
            KeyCode::Down | KeyCode::Char('j') => Some(Action::MenuDown),
            KeyCode::Enter => Some(Action::Confirm),
            KeyCode::Esc => Some(Action::Back),
            _ => None,
        },
        Screen::DiceOff { .. } => match key.code {
            KeyCode::Enter | KeyCode::Char(' ') => Some(Action::Confirm),
            _ => None,
        },
        Screen::Game => match key.code {
            KeyCode::Char(' ') => Some(Action::RollDice),
            KeyCode::Left | KeyCode::Char('h') => Some(Action::SelectPrev),
            KeyCode::Right | KeyCode::Char('l') if !matches!(key.code, KeyCode::Char('l')) => Some(Action::SelectNext),
            KeyCode::Right => Some(Action::SelectNext),
            KeyCode::Char('l') => Some(Action::ToggleLog),
            KeyCode::Up | KeyCode::Char('k') => Some(Action::SelectPrev),
            KeyCode::Down | KeyCode::Char('j') => Some(Action::SelectNext),
            KeyCode::Enter => Some(Action::ConfirmMove),
            KeyCode::Esc => Some(Action::QuitPrompt),
            _ => None,
        },
        Screen::GameOver => match key.code {
            KeyCode::Char('n') | KeyCode::Char('N') => Some(Action::NewGame),
            KeyCode::Char('q') | KeyCode::Char('Q') | KeyCode::Esc => Some(Action::Quit),
            _ => None,
        },
    }
}
```

Note: in `Screen::Game`, `l` toggles log and `→` selects next. These don't conflict because `l` is a distinct keycode from `Right`.

// ur-cli/src/main.rs
mod app;
mod animation;
mod input;
mod ui;

use std::io;
use std::time::Duration;

use crossterm::{
    event::{self, DisableMouseCapture, EnableMouseCapture, Event},
    execute,
    terminal::{disable_raw_mode, enable_raw_mode, EnterAlternateScreen, LeaveAlternateScreen},
};
use ratatui::{backend::CrosstermBackend, Terminal};

use app::App;
use input::{map_key, Action};

fn main() -> io::Result<()> {
    // Setup terminal
    enable_raw_mode()?;
    let mut stdout = io::stdout();
    execute!(stdout, EnterAlternateScreen, EnableMouseCapture)?;
    let backend = CrosstermBackend::new(stdout);
    let mut terminal = Terminal::new(backend)?;

    let mut app = App::new();
    let tick_rate = Duration::from_millis(33); // ~30 fps

    let result = run(&mut terminal, &mut app, tick_rate);

    // Restore terminal
    disable_raw_mode()?;
    execute!(
        terminal.backend_mut(),
        LeaveAlternateScreen,
        DisableMouseCapture,
    )?;
    terminal.show_cursor()?;

    result
}

fn run<B: ratatui::backend::Backend>(
    terminal: &mut Terminal<B>,
    app: &mut App,
    tick_rate: Duration,
) -> io::Result<()> {
    loop {
        let size = terminal.size()?;

        terminal.draw(|f| {
            if size.width < 80 || size.height < 24 {
                ui::render_too_small(f);
            } else {
                ui::render(f, app);
            }
        })?;

        if event::poll(tick_rate)? {
            if let Event::Key(key) = event::read()? {
                if let Some(action) = map_key(key, &app.screen) {
                    handle_action(app, action);
                }
            }
        } else {
            // Tick: advance animations
            animation::tick(app);
        }

        if app.should_quit {
            break;
        }
    }
    Ok(())
}

fn handle_action(app: &mut App, action: Action) {
    match action {
        Action::Quit => app.quit(),
        Action::NewGame => app.start_new_game(),
        Action::Confirm => app.handle_confirm(),
        Action::Back => app.handle_back(),
        Action::MenuUp => app.handle_menu_up(),
        Action::MenuDown => app.handle_menu_down(),
        Action::RollDice => app.handle_roll_dice(),
        Action::SelectPrev => app.handle_select_prev(),
        Action::SelectNext => app.handle_select_next(),
        Action::ConfirmMove => app.handle_confirm_move(),
        Action::ToggleLog => app.log_visible = !app.log_visible,
        Action::QuitPrompt => app.quit(), // simplified: quit immediately
    }
}

// ur-cli/src/ui/title.rs
use ratatui::{
    Frame,
    layout::{Alignment, Constraint, Direction, Layout, Rect},
    style::{Color, Modifier, Style},
    text::{Line, Span},
    widgets::{Block, Clear, Paragraph},
};

const LOGO: &[&str] = &[
    " ██╗   ██╗██████╗ ",
    " ██║   ██║██╔══██╗",
    " ██║   ██║██████╔╝",
    " ██║   ██║██╔══██╗",
    " ╚██████╔╝██║  ██║",
    "  ╚═════╝ ╚═╝  ╚═╝",
];

const CUNEIFORM_BORDER: &str = "𒀭  𒀭  𒀭  𒀭  𒀭  𒀭  𒀭  𒀭";
const CUNEIFORM_SCRIPT: &str = "𒆳𒆳 𒀭𒂗𒍪 𒆳𒆳";

pub fn render(f: &mut Frame, selected: usize) {
    let area = f.size();

    // Centre vertically: total height = border + gap + "The Royal Game of" + logo(6) + cuneiform + subtitle + gap + menu(2) + border
    let total_h = 1 + 1 + 1 + 6 + 1 + 1 + 1 + 2 + 1; // ~15
    let vpad = area.height.saturating_sub(total_h) / 2;

    let chunks = Layout::default()
        .direction(Direction::Vertical)
        .constraints([
            Constraint::Length(vpad),
            Constraint::Min(total_h),
            Constraint::Length(vpad),
        ])
        .split(area);
    let inner = chunks[1];

    let gold = Style::default().fg(Color::Yellow);
    let dim = Style::default().fg(Color::DarkGray);
    let sub = Style::default().fg(Color::Gray);

    let mut lines: Vec<Line> = Vec::new();

    lines.push(Line::from(Span::styled(CUNEIFORM_BORDER, dim)).alignment(Alignment::Center));
    lines.push(Line::from(""));
    lines.push(Line::from(Span::styled("The Royal Game of", sub)).alignment(Alignment::Center));

    for logo_line in LOGO {
        lines.push(Line::from(Span::styled(*logo_line, gold.add_modifier(Modifier::BOLD))).alignment(Alignment::Center));
    }

    lines.push(Line::from(Span::styled(CUNEIFORM_SCRIPT, dim)).alignment(Alignment::Center));
    lines.push(Line::from(Span::styled("circa 2600 BCE · Mesopotamia", sub)).alignment(Alignment::Center));
    lines.push(Line::from(""));

    let menu_items = ["New Game", "Quit"];
    for (i, item) in menu_items.iter().enumerate() {
        let style = if i == selected {
            Style::default().fg(Color::Black).bg(Color::Yellow)
        } else {
            sub
        };
        lines.push(Line::from(Span::styled(format!("[ {} ]", item), style)).alignment(Alignment::Center));
    }

    lines.push(Line::from(""));
    lines.push(Line::from(Span::styled(CUNEIFORM_BORDER, dim)).alignment(Alignment::Center));

    let p = Paragraph::new(lines);
    f.render_widget(p, inner);
}

    widgets::Widget,

use ur_core::{
    board::Square,

/// Colors for the two players.
pub const COLOR_P1: Color = Color::LightBlue;
pub const COLOR_P2: Color = Color::LightRed;
/// Background for rosette tiles.
pub const COLOR_ROSETTE_BG: Color = Color::Rgb(61, 43, 31); // warm amber dark
/// Foreground for empty rosette marker.
pub const COLOR_ROSETTE_FG: Color = Color::Yellow;
/// Background for selected piece.
pub const COLOR_SELECT_BG: Color = Color::Yellow;

/// Renders the board into `area`. Origin is the top-left of the widget.
///
/// `selected_square`: the square of the currently selected piece (highlighted).
/// `target_squares`: squares that are valid move destinations (dimly highlighted).
pub struct BoardWidget<'a> {
    pub rules: &'a GameRules,
    pub board: &'a Board,
    pub selected_square: Option<Square>,
    pub target_squares: &'a [Square],
}

impl<'a> Widget for BoardWidget<'a> {
    fn render(self, area: Rect, buf: &mut Buffer) {
        let rules = self.rules;

        // For each row and valid col, compute screen position.
        // Tile width = 4 chars content + borders handled by drawing chars directly.
        // We draw the board as text into the buffer.

        // Row screen Y offsets: row 0 → y, row 1 → y+1, row 2 → y+2
        // Col screen X offsets depend on column and the gap:
        //   cols 0–3: x_base + col * 5
        //   cols 4–5: only valid for row 1: x_base + col * 5
        //   cols 6–7: x_base + col * 5  (gap takes same space as cols 4–5)

        let x0 = area.x;
        let y0 = area.y;

        // Draw each valid square as a 4-char cell.
        // Borders are drawn separately via box-drawing chars.
        // Simple approach: render each cell at its (row, col) position.

        for sq in rules.board_shape.valid_squares() {
            let cx = x0 + sq.col as u16 * 5;
            let cy = y0 + sq.row as u16;

            if cx + 4 > area.right() || cy >= area.bottom() {
                continue;
            }

            let is_rosette = rules.board_shape.is_rosette(*sq);
            let occupant = self.board.get(*sq);
            let is_selected = self.selected_square == Some(*sq);
            let is_target = self.target_squares.contains(sq);

            // Determine bg
            let bg = if is_selected {
                COLOR_SELECT_BG
            } else if is_rosette {
                COLOR_ROSETTE_BG
            } else if is_target {
                Color::Rgb(20, 40, 20)
            } else {
                Color::Reset
            };

            // Determine content and fg
            let (content, fg) = match occupant {
                Some(piece) => {
                    let color = if piece.player == Player::Player1 { COLOR_P1 } else { COLOR_P2 };
                    let fg = if is_selected { Color::Black } else { color };
                    (" ● ", fg)
                }
                None => {
                    if is_rosette {
                        (" ✦ ", COLOR_ROSETTE_FG)
                    } else if is_target {
                        (" · ", Color::DarkGray)
                    } else {
                        ("   ", Color::Reset)
                    }
                }
            };

            let style = Style::default().fg(fg).bg(bg);
            // Render left border char
            buf.get_mut(cx, cy).set_char('│').set_style(Style::default().fg(Color::DarkGray));
            // Render content (3 chars)
            for (i, ch) in content.chars().enumerate() {
                buf.get_mut(cx + 1 + i as u16, cy).set_char(ch).set_style(style);
            }
        }
    }
}

/// Returns the target square for the given move, if it lands on a board square.
pub fn move_target(mv: &Move) -> Option<Square> {
    match mv.to {
        PieceLocation::OnBoard(sq) => Some(sq),
        _ => None,
    }
}

/// Returns the source square for the given move, if the piece is currently on the board.
pub fn move_source(mv: &Move) -> Option<Square> {
    match mv.from {
        PieceLocation::OnBoard(sq) => Some(sq),
        _ => None,
    }
}

/// Current animation state.
#[derive(Debug)]
pub enum Animation {
    /// Dice rolling — cycles display until frames_remaining hits 0.
    DiceRoll {
        frames_remaining: u32,
        final_value: Dice,
        display: Dice,
    },
    /// Piece moving square by square along a path.
    PieceMove {
        /// Remaining squares in the path.
        remaining: Vec<Square>,
        /// Frames to wait per square.
        frames_per_step: u32,
        frames_this_step: u32,
        /// The piece symbol to animate (player color determined by App).
        is_player1: bool,
    },
    /// Captured piece flashing before disappearing.
    CaptureFlash {
        square: Square,
        frames_remaining: u32,
    },
    /// Dice-off animation: both players' dice cycling.
    DiceOff {
        p1_frames: u32,
        p2_frames: u32,
        p1_final: Dice,
        p2_final: Dice,
        p1_display: Dice,
        p2_display: Dice,
    },
    /// AI is computing — spinner frame.
    AiThinking { frame: u32 },
    /// Animation finished — caller should clear this.
    Done,
}

/// Tick one animation frame. Transitions to `Done` when complete.
pub fn tick_animation(anim: &mut Animation) {
    match anim {
        Animation::DiceRoll { frames_remaining, final_value, display } => {
            if *frames_remaining <= 1 {
                *display = *final_value;
                *anim = Animation::Done;
            } else {
                *frames_remaining -= 1;
                // Cycle display: alternate 0..4
                display.0 = (display.0 + 1) % 5;
            }
        }
        Animation::PieceMove { remaining, frames_per_step, frames_this_step, .. } => {
            if *frames_this_step > 0 {
                *frames_this_step -= 1;
            } else if remaining.is_empty() {
                *anim = Animation::Done;
            } else {
                remaining.remove(0);
                *frames_this_step = *frames_per_step;
                if remaining.is_empty() {
                    *anim = Animation::Done;
                }
            }
        }
        Animation::CaptureFlash { frames_remaining, .. } => {
            if *frames_remaining <= 1 {
                *anim = Animation::Done;
            } else {
                *frames_remaining -= 1;
            }
        }
        Animation::DiceOff { p1_frames, p2_frames, p1_final, p2_final, p1_display, p2_display } => {
            if *p1_frames > 0 {
                *p1_frames -= 1;
                p1_display.0 = (p1_display.0 + 1) % 5;
                if *p1_frames == 0 { *p1_display = *p1_final; }
            }
            if *p2_frames > 0 {
                *p2_frames -= 1;
                p2_display.0 = (p2_display.0 + 1) % 5;
                if *p2_frames == 0 { *p2_display = *p2_final; }
            }
        }
        Animation::AiThinking { frame } => {
            *frame = (*frame + 1) % 4;
        }
        Animation::Done => {}
    }
}

/// Advance app animations and handle post-animation transitions.
pub fn tick(app: &mut App) {
    if let Some(anim) = &mut app.animation {
        tick_animation(anim);
        if matches!(anim, Animation::Done) {
            app.animation = None;
            app.on_animation_done();
        }
    }
    // Advance AI thinking spinner even when waiting for thread
    if app.ai_thinking {
        app.ai_spinner_frame = (app.ai_spinner_frame + 1) % 4;
        app.poll_ai_move();
    }
}

    self.dice_roll = Some(final_value); // stored for after animation
}

pub fn handle_select_prev(&mut self) {
    if self.legal_moves.is_empty() { return; }
    if self.selected_move_idx == 0 {
        self.selected_move_idx = self.legal_moves.len() - 1;
    } else {
        self.selected_move_idx -= 1;
    }
}

pub fn handle_select_next(&mut self) {
    if self.legal_moves.is_empty() { return; }
    self.selected_move_idx = (self.selected_move_idx + 1) % self.legal_moves.len();
}

pub fn handle_confirm_move(&mut self) {
    if self.animation.is_some() { return; }
    let mv = match self.legal_moves.get(self.selected_move_idx) {
        Some(m) => m.clone(),
        None => return,
    };
    self.apply_move(mv);
}

/// Applies a move, updates stats, starts animations.
pub fn apply_move(&mut self, mv: ur_core::state::Move) {
    let gs = match &self.game_state {
        Some(gs) => gs.clone(),
        None => return,
    };
    let result = gs.apply_move(mv.clone());
    self.stats.moves += 1;

    // Log the move
    let piece_desc = format!("P{}", mv.piece.player.index() + 1);
    match &mv.to {
        ur_core::state::PieceLocation::OnBoard(sq) => {
            if let Some(captured) = &result.captured {
                self.stats.captures[mv.piece.player.index()] += 1;
                self.log.push(format!("{} captured on ({},{})", piece_desc, sq.row, sq.col));
            } else if result.landed_on_rosette {
                self.log.push(format!("{} landed on rosette ({},{}) — extra turn!", piece_desc, sq.row, sq.col));
            } else {
                self.log.push(format!("{} moved to ({},{})", piece_desc, sq.row, sq.col));
            }
        }
        ur_core::state::PieceLocation::Scored => {
            self.log.push(format!("{} scored a piece!", piece_desc));
        }
        _ => {}
    }

    // Apply the move immediately. Piece movement and capture flash animations
    // are defined in the Animation enum (Task 8) but not yet wired here —
    // the game is fully playable without them. Wire them after the base game
    // is working end-to-end.
    self.game_state = Some(result.new_state.clone());
    self.dice_roll = None;
    self.legal_moves.clear();
    self.selected_move_idx = 0;

    if result.game_over {
        self.screen = crate::app::Screen::GameOver;
        return;
    }

    // If AI's turn now, start AI
    if result.new_state.current_player == ur_core::player::Player::Player2 {
        self.start_ai_turn();
    }
}

/// Called when a dice-roll animation completes.
pub fn on_animation_done(&mut self) {
    // If a dice roll just finished, compute legal moves
    if let Some(roll) = self.dice_roll {
        if let Some(gs) = &self.game_state {
            let moves = gs.legal_moves(roll);
            if moves.is_empty() {
                // No moves — forfeit turn
                self.log.push(format!("Roll {} — no moves, turn forfeited", roll.value()));
                let new_state = gs.clone();
                // Advance to next player by applying a forfeit (ur-core handles this via GameState::advance_turn)
                // Since legal_moves returns empty, just flip the player manually isn't correct.
                // Instead call apply_move is not possible with no moves.
                // ur-core's contract: if legal_moves is empty, caller forfeits. We need a skip-turn path.
                // Use GameState::forfeit_turn() — see note below.
                if let Some(new_gs) = gs.forfeit_turn() {
                    self.game_state = Some(new_gs.clone());
                    if new_gs.current_player == ur_core::player::Player::Player2 {
                        self.start_ai_turn();
                    }
                }
                self.dice_roll = None;
            } else {
                self.legal_moves = moves;
                self.selected_move_idx = 0;
            }
        }
    }
}
```

impl GameState {
    /// Advances to the next player's turn without applying a move.
    /// Used when `legal_moves` returns empty (roll of 0 or no valid squares).
    /// Returns `None` if the game is already over.
    pub fn forfeit_turn(&self) -> Option<Self> {
        match &self.phase {
            GamePhase::GameOver(_) => None,
            GamePhase::WaitingForRoll => {
                let mut next = self.clone();
                next.current_player = self.current_player.opponent();
                Some(next)
            }
        }

use std::sync::mpsc;
use std::thread;

/// Starts the AI's turn: rolls dice, then either forfeits (roll=0) or
/// spawns a thread to compute the best move.
pub fn start_ai_turn(&mut self) {
    let gs = match &self.game_state {
        Some(gs) => gs.clone(),
        None => return,
    };
    let roll = ur_core::dice::Dice::roll(&mut self.rng);
    self.log.push(format!("AI rolled {}", roll.value()));

    let moves = gs.legal_moves(roll);
    if moves.is_empty() {
        self.log.push("AI roll — no moves, turn forfeited".to_string());
        if let Some(new_gs) = gs.forfeit_turn() {
            self.game_state = Some(new_gs);
        }
        return;
    }

    let depth = self.difficulty;
    let (tx, rx) = mpsc::channel();
    self.ai_receiver = Some(rx);
    self.ai_thinking = true;
    self.dice_roll = Some(roll);

    thread::spawn(move || {
        let chosen = ur_core::ai::choose_move(&gs, roll, depth);
        let _ = tx.send(chosen);
    });
}

/// Checks if the AI thread has returned a move and applies it.
pub fn poll_ai_move(&mut self) {
    let mv = match &self.ai_receiver {
        Some(rx) => match rx.try_recv() {
            Ok(m) => m,
            Err(_) => return,
        },
        None => return,
    };
    self.ai_receiver = None;
    self.ai_thinking = false;
    self.apply_move(mv);
}
```

pub fn tick(app: &mut App) {

            app.on_animation_done();

    if let crate::app::Screen::DiceOff { state } = &mut app.screen {
        if state.p1_frames > 0 {
            state.p1_frames -= 1;
            state.p1_display.0 = (state.p1_display.0 + 1) % 5;
            if state.p1_frames == 0 { state.p1_display = state.p1_final; }
        }
        if state.p2_frames > 0 {
            state.p2_frames -= 1;
            state.p2_display.0 = (state.p2_display.0 + 1) % 5;
            if state.p2_frames == 0 { state.p2_display = state.p2_final; }
        }
        // Determine winner once both done
        if state.p1_frames == 0 && state.p2_frames == 0 && state.winner.is_none() {
            use ur_core::player::Player;
            state.winner = match state.p1_final.0.cmp(&state.p2_final.0) {
                std::cmp::Ordering::Greater => Some(Player::Player1),
                std::cmp::Ordering::Less => Some(Player::Player2),
                std::cmp::Ordering::Equal => None, // tie — re-roll after short pause
            };
            // On tie: schedule re-roll after a brief pause (handled next tick)
        }
    }

    // AI spinner
    if app.ai_thinking {
        app.ai_spinner_frame = (app.ai_spinner_frame + 1) % 4;
        app.poll_ai_move();
    }
